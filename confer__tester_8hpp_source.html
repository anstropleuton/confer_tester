315  {
316  std::vector<std::pair<const test *, std::size_t>> failed_tests = {};
317  for (auto &test : tests)
318  {
319  if (pre_run) pre_run(test);
320  std::size_t errors = test->run();
321  bool end_test = false;
322 
323  if (errors != 0)
324  {
325  failed_tests.emplace_back(test, errors);
326  if (run_failed && run_failed(test, errors))
327  {
328  end_test = true;
329  }
330  }
331 
332  if (post_run) post_run(test, errors);
333  if (end_test) break;
334  }
335  return failed_tests;
336  };
